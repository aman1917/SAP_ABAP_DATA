at selection-screen on radiobutton group grp1.
  case sy-ucomm.
    when 'FC1'.
    if p_add = 'X'.
      v_z = p_x + p_y.
      lv_str1 = 'Sum of Two Numbers is'.
      lv_str2 = v_z.
      clear lv_msg.
      concatenate lv_str1 lv_str2 into lv_msg separated by space.
      message lv_msg type 'I'.

    elseif p_sub = 'X'.
      v_z = p_x - p_y.
      if v_z >= 0.
        lv_str1 = 'Difference of Two Numbers is'.
        lv_str2 = v_z.
        clear lv_msg.
        concatenate lv_str1 lv_str2 into lv_msg separated by space.
        message lv_msg type 'I'.

      else.
        lv_str1 = 'Difference of Two Numbers is -'.
        lv_str2 = v_z.
        clear lv_temp.
        concatenate lv_str1 lv_str2 into lv_temp separated by space.
        lv_len = strlen( lv_temp ).
        lv_pos = lv_len - 1.
        clear lv_msg.
        lv_msg = lv_temp+0(lv_pos).
        message lv_msg type 'I'.

      endif.

     elseif p_mul = 'X'.
      v_z = p_x * p_y.
      lv_str1 = 'Sum of Two Numbers is'.
      lv_str2 = v_z.
      clear lv_msg.
      concatenate lv_str1 lv_str2 into lv_msg separated by space.
      message lv_msg type 'I'.

     elseif p_div = 'X'.
       if p_y = 0.
         message 'Divisible By ZERO!!!' type 'I'.
       else.
         lv_z = p_x / p_y.
         lv_str1 = 'Division of Two Numbers is'.
         lv_str2 = lv_z.
         clear lv_msg.
         concatenate lv_str1 lv_str2 into lv_msg separated by space.
         message lv_msg type 'I'.
       endif.
     else.
       message 'None Of Radio Button Selected!!!' type 'I'.

     endif.
endcase.
