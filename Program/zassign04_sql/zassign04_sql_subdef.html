FORM invisible .
LOOP AT SCREEN.
    if
     screen-group1 = 'ID1' .
      screen-input = '0'.
*      screen-output = '0'.
      screen-invisible = '1'.
     modify screen.
     endif.
    endloop.
ENDFORM.

FORM fetch .
  select SINGLE stud_id,stud_name,stud_batch,stud_course,attendance,stud_date
    FROM zstd_course as s
    LEFT JOIN zstd_attend as a
    ON s~stud_id = a~std_id
    where s~stud_id = @p_stdid
    into @wa_det.
ENDFORM.

FORM setvalues .
  p_stdid = wa_det-stud_id.
  p_stdnm = wa_det-stud_name.
  p_batch = wa_det-stud_batch.
  p_course = wa_det-stud_course.
  p_attend = wa_det-attendance.
  p_date = wa_det-stud_date.
ENDFORM.
