Description: Selection Option
REPORT ZMAKE_PROGRAM25.

data v_netwr type vbak-netwr.
select-options so_netwr for v_netwr.

types: begin of ty_sales,
        vbeln type vbak-vbeln,
        erdat type vbak-erdat,
        erzet type vbak-erzet,
        ernam type vbak-ernam,
        netwr type vbak-netwr,
       end of ty_sales.

data: lt_sales type table of ty_sales,
      wa_sales type ty_sales.

start-of-selection.
  perform getsalesorder.
  if lt_sales is not initial.
    describe table lt_sales.
    write:/ 'No. of sales order in given range :'.
    perform displaysalesorder.
  else.
    message 'No sales order in given net value range' type 'I'.
  endif.

FORM getsalesorder .
  select vbeln erdat erzet ernam netwr
    from vbak
    into table lt_sales
    where netwr in so_netwr.
ENDFORM.

FORM displaysalesorder .
  loop at lt_sales into wa_sales.
    write:/ wa_sales-vbeln,
            wa_sales-erdat,
            wa_sales-ernam,
            wa_sales-netwr.
  endloop.
ENDFORM.
