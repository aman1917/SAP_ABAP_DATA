PROGRAM ZSTD_PROG6.
*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*\
*TYPE-POOLS vrm.

DATA gv_flag TYPE i.

DATA: lb1(2) TYPE c,
     c1 TYPE c,
     c2 TYPE c,
     c3 TYPE c,
     r1 TYPE c,
     r2 TYPE c,
     r3 TYPE c.

DATA: lt_values TYPE vrm_values,
      wa_values LIKE LINE OF lt_values.

MODULE user_command_0100 INPUT.
  CASE sy-ucomm.
    WHEN 'FC1'.
      LEAVE PROGRAM.
    WHEN 'FC2'.
      IF lb1 = 'K1'.
        gv_flag = 2.
      ELSEIF lb1 = 'K2'.
        gv_flag = 3.
      ELSEIF lb1 = 'K3'.
        gv_flag = 4.
      ENDIF.

    WHEN 'FC3'.
      IF c1 = 'X'.
        MESSAGE 'CheckBox 1 Selected.' TYPE 'I'.
      ENDIF.
    WHEN 'FC4'.
      IF c2 = 'X'.
        MESSAGE 'CheckBox 2 Selected.' TYPE 'I'.
      ENDIF.
    WHEN 'FC5'.
      IF c3 = 'X'.
        MESSAGE 'CheckBox 3 Selected.' TYPE 'I'.
      ENDIF.
    WHEN 'FC6'.
      IF r1 = 'X'.
        MESSAGE 'RadioBox 1 Selected.' TYPE 'I'.
      ELSEIF r2 = 'X'.
        MESSAGE 'RadioBox 2 Selected.' TYPE 'I'.
      ELSEIF r3 = 'X'.
        MESSAGE 'RadioBox 3 Selected.' TYPE 'I'.
      ENDIF.
  ENDCASE.
ENDMODULE.

MODULE status_0100 OUTPUT.
* SET PF-STATUS 'xxxxxxxx'.
* SET TITLEBAR 'xxx'.
  IF gv_flag eq 0.
    gv_flag = 1.
    PERFORM preparevalues.
    PERFORM invisible.
  ELSEIF gv_flag eq 2.
    PERFORM visible1.
  ELSEIF gv_flag eq 3.
    PERFORM visible2.
  ELSEIF gv_flag eq 4.
    PERFORM visible3.
  ENDIF.
ENDMODULE.
