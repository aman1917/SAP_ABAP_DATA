types: begin of ty_emp,
        empno type i,
        ename(10) type c,
        empdesig(15) type c,
        end of ty_emp.

data ls_emp type table of ty_emp.

data wa_emp type ty_emp.

clear wa_emp.
  wa_emp-empno = 1.
  wa_emp-ename = 'Aman'.
  wa_emp-empdesig = 'Manager'.
 append wa_emp to ls_emp.

clear wa_emp.
  wa_emp-empno = 2.
  wa_emp-ename = 'Saloni'.
  wa_emp-empdesig = 'Director'.
 append wa_emp to ls_emp.

 clear wa_emp.
  wa_emp-empno = 3.
  wa_emp-ename = 'Pranali'.
  wa_emp-empdesig = 'CEO'.
 append wa_emp to ls_emp.

 write:/ 'Data In Internal Table'.
 loop at ls_emp into wa_emp.
 write:/ wa_emp-empno,wa_emp-ename,wa_emp-empdesig.
 endloop.

 write:/ 'Data In Internal Table'.
 loop at ls_emp into wa_emp where empdesig = 'Manager'.
 write:/ wa_emp-empno,wa_emp-ename,wa_emp-empdesig.
 endloop.

clear wa_emp.
  wa_emp-empno = 4.
  wa_emp-ename = 'Amruta'.
  wa_emp-empdesig = 'Cp-founder'.
 insert wa_emp into ls_emp index 2.

 format color 3.
  write:/ 'Data In Internal Table'.
 loop at ls_emp into wa_emp.
 write:/ wa_emp-empno,wa_emp-ename,wa_emp-empdesig.
 endloop.
 format color off.

 uline.
 describe table ls_emp.
 write:/' Number Record in thae table is ',sy-tfill.

 clear wa_emp.

 data ls_emp2 type table of ty_emp with non-unique key empno.

 append lines of ls_emp to ls_emp2.

sort ls_emp2.
  format color 2.

 write:/ 'Data In Internal Table emp2'.
 loop at ls_emp2 into wa_emp.
 write:/ wa_emp-empno,wa_emp-ename,wa_emp-empdesig.
 endloop.

 format color off.

uline.
 sort ls_emp2 by empdesig.
  format color 5.

 write:/ 'Data In Internal Table emp2'.
 loop at ls_emp2 into wa_emp.
 write:/ wa_emp-empno,wa_emp-ename,wa_emp-empdesig.
 endloop.

 uline.
 sort ls_emp2 by empdesig descending.
  format color 5.

 write:/ 'Data In Internal Table emp2'.
 loop at ls_emp2 into wa_emp.
 write:/ wa_emp-empno,wa_emp-ename,wa_emp-empdesig.
 endloop.

 ls_emp2 = ls_emp.

 format color off.

  uline.
 describe table ls_emp.
 write:/' Number Record in thae table is ',sy-tfill.

 uline.
 clear ls_emp.
 read table ls_emp into wa_emp index 2.
 if sy-subrc eq 0.
  write:/ 'Record Found in emp2'.
  write:/ wa_emp-empno,wa_emp-ename,wa_emp-empdesig.
 else.
  write:/ 'Record Not Found in emp2'.
 endif.

 uline.
  clear ls_emp2.
 read table ls_emp2 into wa_emp index 2 transporting empno ename.
 if sy-subrc eq 0.
  write:/ 'Record Found in emp2'.

 else.
  write:/ 'Record Not Found in emp2'.
 endif.
