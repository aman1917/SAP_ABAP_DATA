*&---------------------------------------------------------------------*
*& Include ZSTD_PROG3_TOP                           - Module Pool      ZSTD_PROG3
*&---------------------------------------------------------------------*
PROGRAM ZSTD_PROG3.

CONTROLS tbstr TYPE TABSTRIP.

DATA v_scrno TYPE sy-dynnr value '102'.

DATA v_flag TYPE i.

MODULE user_command_0100 INPUT.
  CASE sy-ucomm.
    WHEN 'FC1'.
      LEAVE PROGRAM.
    WHEN 'FC2'.
      tbstr-activetab = 'FC2'.
      v_scrno = '101'.
    WHEN 'FC3'.
      tbstr-activetab = 'FC3'.
      v_scrno = '102'.
  ENDCASE.
ENDMODULE.

MODULE abc OUTPUT.
* SET PF-STATUS 'xxxxxxxx'.
* SET TITLEBAR 'xxx'.\
  if v_flag eq 0.
    v_flag = 1.
    tbstr-activetab = 'FC3'.
  ENDIF.
ENDMODULE.
