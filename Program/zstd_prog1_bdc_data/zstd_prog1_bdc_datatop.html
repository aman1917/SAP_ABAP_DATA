PROGRAM ZSTD_PROG1_BDC_DATA.

TABLES zstd_course.

CONTROLS tbctrl TYPE TABLEVIEW USING SCREEN 100.

TYPES: BEGIN OF ty_std,
        stud_id TYPE zstd_course-stud_id,
        stud_name TYPE zstd_course-stud_name,
        stud_batch TYPE zstd_course-stud_batch,
        stud_course TYPE zstd_course-stud_course,
  END OF ty_std.

DATA: lt_std TYPE TABLE OF ty_std,
      wa_std TYPE ty_std.

MODULE user_command_0100 INPUT.
  CASE sy-ucomm.
    WHEN 'FC1'.
      CALL TRANSACTION 'ZSTD_BDC_CALL'.
  ENDCASE.
ENDMODULE.

MODULE exit INPUT.
  CASE sy-ucomm.
    WHEN 'FC2'.
      LEAVE PROGRAM.
  ENDCASE.
ENDMODULE.

MODULE getdata OUTPUT.
  SELECT stud_id stud_name stud_batch stud_course
    FROM zstd_course
    INTO TABLE lt_std.
ENDMODULE.

MODULE transferitemdata OUTPUT.
  clear zstd_course.
  zstd_course-stud_id = wa_std-stud_id.
  zstd_course-stud_name = wa_std-stud_name.
  zstd_course-stud_batch = wa_std-stud_batch.
  zstd_course-stud_course = wa_std-stud_course.
ENDMODULE.
