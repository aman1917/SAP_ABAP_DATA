FORM fetdata .
  SELECT SINGLE emp_no emp_name emp_des join_date join_time
    FROM zmake_check
    into wa_check
    where emp_no = p_empno.
ENDFORM.

FORM setvalues .
  p_empnam = wa_check-emp_name.
  p_empdes = wa_check-emp_des.
  p_empdat = wa_check-join_date.
  p_empjoi = wa_check-join_time.
ENDFORM.

FORM invisible .
  LOOP AT SCREEN.
    if
     screen-group1 = 'ID1' or
      screen-name = 'BUT4' or
      screen-name = 'BUT5' .
      screen-input = '0'.
      screen-output = '0'.
      screen-invisible = '1'.
     modify screen.
     endif.
    endloop.
ENDFORM.


FORM cleardata .
  clear p_empno.
ENDFORM.

FORM getf4value .
  select emp_no FROM zmake_check
    into table lt_getf4value.
ENDFORM.

FORM delete_op .
delete FROM zmake_check where emp_no = p_empno.
ENDFORM.



FORM update_data .
UPDATE zmake_check set emp_name = @p_empnam , emp_des = @p_empdes,
                       join_date = @p_empdat, join_time = @p_empjoi
        where emp_no = @p_empno.
ENDFORM.
