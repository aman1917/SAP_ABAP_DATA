start-of-selection.
      if p_r1 = 'X'.
        perform getMaterial.
        if lt_ty_data is not initial.
          perform displayMaterial.
        else.
          message 'No Data Found' type 'I'.
        endif.

      elseif p_r2 = 'X'.
        perform getMonthMaterial.
        if lt_ty_data is not initial.
          perform displayMonthMaterial.
        else.
          message 'No Data Found' type 'I'.
        endif.

      elseif p_r3 = 'X'.
        perform getCustomerMaterial.
        if lt_ty_data is not initial.
          perform displayCustomerMaterial.
        else.
          message 'No Data Found' type 'I'.
        endif.

      else.
           message 'None of them Selected' type 'I'.
      endif.
*endcase.

FORM getMaterial .
    select vbeln posnr matnr matkl erdat
      from vbap
     into corresponding fields of table lt_ty_data
      where erdat in s_erdat.

ENDFORM.

FORM displayMaterial .
  sort lt_ty_data by vbeln.
  loop at lt_ty_data into wa_ty_data.
    write:/ wa_ty_data-vbeln,
            wa_ty_data-posnr,
            wa_ty_data-matnr,
            wa_ty_data-matkl,
            wa_ty_data-erdat.
   endloop.
ENDFORM.
